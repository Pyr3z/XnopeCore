  <Xnope.Defs.SpawnCategoryInjectorDef>
    <defName>SCI_unique_name</defName>
    
    <newCategory>newCategory</newCategory>
	
	<injectToBackstories>
	  <li>full_title_of_backstory_1</li>
	  <li>full_title_of_backstory_2</li>
	</injectToBackstories>
  </Xnope.Defs.SpawnCategoryInjectorDef>
